public class EstruturaDeRepeticao {

  public static void main (String [] args){
  
    //loop infinito
      for (;;) {
       System.out.println("x");
      }



     //imprimi de 1 a 10
      for (int x = 1; x <= 10; x++){
        System.out.println(x);
      }


     //imprimi de 10 a 1
      for (int y = 10; y > 0; y--){
        System.out.println(y);
      }



  }
}
